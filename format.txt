			char[sizeOfString] subItemTitle
			int sizeOfItem
			int itemOffset
			
			IF (this is the second Subgroup):
				int sizeOfString
				char[sizeOfString] presenterTitle

int magicNumber

int numberOfModels
int UNKNOWN
#SECTION[numberOfModels] Model
	int UNKNOWN
	int numberOfParts
	short numberOfVerts
	short numberOfNormals
	short numberOfCoordinates
	short UNKNOWN
	#SECTION[numberOfVerts] Vertice
		float x
		float y
		float z
	#SECTION[numberOfNormals / 2] Normal
		float x
		float y
		float z
	#SECTION[numberOfCoordinates] Coordinate
		float u
		float v
	#SECTION[numberOfParts] Part
		short numTris
		short numIndexDefinitions
		#SECTION[numTris] Indice
			#SECTION[3] Axis
				unsigned short first
				unsigned short second
				//if second has its most significant bit set to 1, then this component
				//of the Indice is an index definition. this means it will be referenced
				//by other Indice components that have a second with its MSB as 0. once it
				//has been determined wheter or not this is a definition, the MSB of second
				//should be set to 0, and then first is either an index to a Vertice defined
				//above, or an index to an index of a Vertice if this component is not an index
				//definition. second is the same as first, except instead of being an index to
				//Vertices it is an index to Normals previously defined.
		int numCoordinateIndexes
		#SECTION[numCoordinateIndexes / 3] CoordinateIndex
			#SECTION[3] Axis
				int coordinateIndex
				//this needs to reference the Indice components from before. Since every Indice
				//has a texture coordinate associated with it, it will reference Indices with the
				//same index as this one. whether or not the corresponding Indice component
				//was an index definition will determine whether or not this CoordinateIndex component
				//is an index definition or not.
				
		byte[1] partColorRed
		byte[1] partColorGreen
		byte[1] partColorBlue
		
		int UNKNOWN
		int UNKNOWN
		
		int sizeOfString
		char[sizeOfString] textureName

		#SECTION[width] Pixel
			//the color index at this pixel is an index to the colors defined in the previous section.
			byte[1] colorIndex
